<script>
let ayahs=[],currentIndex,showingNext=false,score=0,streak=0;

async function loadData(){
  const ar = await fetch("https://api.alquran.cloud/v1/juz/30/ar.alafasy").then(r=>r.json());
  const en = await fetch("https://api.alquran.cloud/v1/juz/30/en.sahih").then(r=>r.json());

  ayahs = ar.data.ayahs.map((a,i)=>({
    text:a.text,
    next:ar.data.ayahs[i+1]?.text || "",
    audio:ar.data.ayahs[i+1]?.audio || "",
    trans:en.data.ayahs[i+1]?.text || "",
    surah:a.surah.englishName,
    num:a.numberInSurah
  }));
}

function showAyah(){
  currentIndex=Math.floor(Math.random()*(ayahs.length-1));
  showingNext=false;
  render();
}

function showNext(){
  showingNext=true;
  render();
  playAudio();
}

function render(){
  const a=ayahs[currentIndex];
  const el=document.getElementById("ayah");

  el.classList.remove("fade");
  void el.offsetWidth;
  el.classList.add("fade");

  if(!showingNext){
    el.innerText=a.text;
    document.getElementById("meta").innerText="";
    document.getElementById("translation").innerText="";
  }else{
    el.innerText=a.next;
    document.getElementById("meta").innerText=a.surah+" • Ayah "+(a.num+1);
    document.getElementById("translation").innerText=a.trans;
  }
}

function playAudio(){
  if(!showingNext) return;
  const audio=document.getElementById("audio");
  audio.src=ayahs[currentIndex].audio;
  audio.play();
}

function markCorrect(){
  score++; streak++;
  updateScore(); showAyah();
}

function markWrong(){
  streak=0;
  updateScore(); showAyah();
}

function updateScore(){
  document.getElementById("score").innerText=score;
  document.getElementById("streak").innerText=streak;
}

function toggleDark(){
  document.body.classList.toggle("dark");
}

loadData().then(()=>{
  document.getElementById("ayah").innerText="Click Start";
});
</script>
